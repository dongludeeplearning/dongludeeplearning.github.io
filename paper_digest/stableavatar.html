class TimestepAwareAdapter(nn.Module):
  def forward(self, audio_emb, latents, t):
    # a. Cross-Attend Audio (Query) with Latents (Key/Value)
    audio_context = self.cross_attn(query=audio_emb, kv=latents)
    
    # b. Modulate with Timestep to inject diffusion priors
    t_emb = self.time_embedding(t)
    return self.ada_ln(audio_context, t_emb)

def predict_noise(model, latents, t, audio, ref_img):
  # Instead of CFG (unconditional), steer towards joint audio-latent distribution
  audio_refined = model.adapter(audio, latents, t)
  noise_pred = model.dit_backbone(latents, t, audio_refined, ref_img)
  return noise_pred # Directly guided by refined audio

def generate_long_video(full_audio):
  clips = []
  for window in split_audio(full_audio, overlap=True):
    # Generate current clip latents
    latents = diffusion_loop(window)
    
    # Fuse overlapping regions with logarithmic weights for smoothness
    if clips:
      latents = fuse_overlap(clips[-1], latents, strategy="log_weight")
    clips.append(latents)
  return decode_video(clips)
